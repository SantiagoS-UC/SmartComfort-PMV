210void setupStateMachine() {
211 State* Monitor = stateMachine.AddState("Monitor", []() {
212 mostrarMensaje("Monitoreando...");

214 State* pmv_alto = stateMachine.AddState("pmv_alto", []() {

219 State* pmv_bajo = stateMachine.AddState("pmv_bajo", []() {
220 mostrarMensaje("Activando calefactor");
221 digitalWrite(pinVentilador, LOW);
222 digitalWrite(pinCalefactor, HIGH);
223 });
224
225 // --- Transiciones --- //
226 stateMachine.AddTransition(Monitor, pmv_alto, []() {
227 return input == pmv && pmv_actual > 1.0f;
228 });
229 stateMachine.AddTransition(Monitor, pmv_bajo, []() {
230 return input == pmv && pmv_actual < -1.0f;
231 });
232 stateMachine.AddTransition(pmv_alto, Monitor, []() {
233 return pmv_actual <= 1.0f;
234 });
235 stateMachine.AddTransition(pmv_bajo, Monitor, []() {
236 return input == tiempo || pmv_actual >= -1.0f;
